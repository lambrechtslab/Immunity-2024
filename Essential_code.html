data <- object[[]] %>% group_by(Patient, Treatment, T_expansion, ann) %>%
                    filter(DataType == "5prime",

                           Treatment == "Durvalumab") %>%
                    summarise(n = n()) %>%
                    mutate(freq = n/sum(n)) %>%
                    ungroup() %>%
                    complete(ann, nesting (Patient,Treatment, T_expansion),
                             fill = list(n = 0, freq = 0)) %>%
                    select(Treatment, Patient, T_expansion, ann, freq) %>%
                    pivot_wider(names_from=ann, values_from=freq)
